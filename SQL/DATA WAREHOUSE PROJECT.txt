# CREATE CUSTOMER DIMENSION TABLLE AND ADD PRIMARY KEY
SELECT DISTINCT CUSTOMER_ID,CUSTOMER_NAME,SEGMENT INTO DIM_CUSTOMER FROM SUPERSTORE_3;
ALTER TABLE DIM_CUSTOMER 
ALTER COLUMN CUSTOMER_ID VARCHAR(20) NOT NULL;
ALTER TABLE DIM_CUSTOMER ADD CONSTRAINT PK_CUSTOMER PRIMARY KEY (CUSTOMER_ID);
# CREATE PRODUCT DIMENSION TABLE AND ADD PRIMARY KEY
SELECT DISTINCT PRODUCT_ID,PRODUCT_NAME,CATEGORY,SUB_CATEGORY INTO DIM_PRODUCT FROM SUPERSTORE_3;
ALTER TABLE DIM_PRODUCT
ALTER COLUMN PRODUCT_ID VARCHAR(30) NOT NULL;
SELECT PRODUCT_ID, COUNT(*) AS COUNT FROM DIM_PRODUCT
GROUP BY PRODUCT_ID HAVING COUNT(*) > 1;
WITH CTE AS ( SELECT *, ROW_NUMBER() OVER ( PARTITION BY
PRODUCT_ID ORDER BY PRODUCT_NAME) AS RN FROM DIM_PRODUCT)
DELETE FROM CTE WHERE RN >1;
SELECT PRODUCT_ID, COUNT(*) FROM DIM_PRODUCT
GROUP BY PRODUCT_ID HAVING COUNT(*) > 1;
ALTER TABLE DIM_PRODUCT ADD CONSTRAINT PK_PRODUCT PRIMARY KEY (PRODUCT_ID);
# CREATE REGION DIMENSION AND ADD PRIMARY KEY
SELECT DISTINCT COUNTRY,STATE,CITY,POSTAL_CODE,REGION INTO DIM_REGION FROM SUPERSTORE_3;
DELETE FROM DIM_REGION WHERE POSTAL_CODE IS NULL;
ALTER TABLE DIM_REGION
ALTER COLUMN POSTAL_CODE VARCHAR(20) NOT NULL;
SELECT POSTAL_CODE,COUNT(*) AS COUNT FROM DIM_REGION
GROUP BY POSTAL_CODE HAVING COUNT(*)>1;
SELECT * FROM DIM_REGION WHERE POSTAL_CODE = '92024';
WITH CTE AS( SELECT *, ROW_NUMBER() OVER (PARTITION BY 
POSTAL_CODE ORDER BY COUNTRY) AS RN FROM DIM_REGION)
DELETE FROM CTE WHERE RN>1;
ALTER TABLE DIM_REGION
ADD CONSTRAINT PK_REGION PRIMARY KEY (POSTAL_CODE);
# CREATE DATE DIMENSION AND ADD PRIMARY KEY

#  TOTAL REVENUE
SELECT SUM(SALES) AS TOTAL_REVENUE FROM FACT_SALES;
#  TOTAL PROFIT
SELECT SUM(PROFIT) AS TOTAL_PROFIT FROM FACT_SALES;
# TOP 5 CUSTOMERS
SELECT TOP 5 CUSTOMER_NAME, SUM(F.SALES) AS TOTAL_SPENT
FROM FACT_SALES F JOIN DIM_CUSTOMER C
ON F.CUSTOMER_ID = C.CUSTOMER_ID
GROUP BY C.CUSTOMER_NAME
ORDER BY TOTAL_SPENT DESC;
# CATEGORY WISE REVENUE
SELECT P.CATEGORY, SUM(F.SALES) AS REVENUE
FROM FACT_SALES F JOIN DIM_PRODUCT P
ON F.PRODUCT_ID = P.PRODUCT_ID
GROUP BY P.CATEGORY;
# MONTHLY SALES TREND
SELECT D.YEAR,D.MONTH,SUM(F.SALES) AS REVENUE
FROM FACT_SALES F JOIN DIM_DATE D
ON F.ORDER_DATE = D.ORDER_DATE
GROUP BY D.YEAR,D.MONTH
ORDER BY D.YEAR,D.MONTH;
# REGION WISE SALE
SELECT R.REGION, SUM(F.SALES) AS REVENUE
FROM FACT_SALES F JOIN DIM_REGION R
ON F.POSTAL_CODE = R.POSTAL_CODE
GROUP BY R.REGION;
# TOP 5 CUSTOMERS
SELECT * FROM ( SELECT C.CUSTOMER_NAME,
SUM(F.SALES) AS TOTAL_SALES,
RANK() OVER (ORDER BY SUM(F.SALES) DESC) AS RANK
FROM FACT_SALES F JOIN DIM_CUSTOMER C
ON F.CUSTOMER_ID = C.CUSTOMER_ID
GROUP BY C.CUSTOMER_NAME)T WHERE RANK <= 5;
# ADD ONE INSIGHT 
SELECT P.CATEGORY, SUM(F.PROFIT) AS TOTAL_PROFIT,
SUM(F.SALES) AS TOTAL_SALES,
(SUM(F.PROFIT)*100.0/SUM(F.SALES)) AS PROFIT_MARGIN
FROM FACT_SALES F JOIN DIM_PRODUCT P
ON F.PRODUCT_ID = P.PRODUCT_ID GROUP BY P.CATEGORY;
